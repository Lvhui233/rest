ajax传输数据

	$.ajax({
		url:"list.php",//数据发送地址
		type: 'post',//数据传输方式
		data:{lanmu:1},//需要传送的数据
		dataType:'json',//返回的数据格式
		success:function(data){
//			遍历数组
			$.each(data,function(k,v){
//				将获得的对象中的信息重组成新内容
				var str = '<li><img src="'+v.imgUrl+'"/><h2>'+v.title+'</h2><p>'+v.desc+'</p></li>';
				$("#news").append(str);
			})
		}
	})
	函数中的两个参数，k代表对象在数组中序号，v代表相应的对象
